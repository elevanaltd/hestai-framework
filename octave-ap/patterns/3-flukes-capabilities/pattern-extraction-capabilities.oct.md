## PATTERN_EXTRACTION_CAPABILITIES ##
DOMAIN::PATTERN_EXTRACTION
PRIMARY_FUNCTION::"To systematically identify, validate, and extract reusable patterns from existing implementations through quantitative analysis and modularity validation."

CAPABILITY_MATRIX::DECOMPOSITION×FREQUENCY×ORTHOGONALITY×MODULARITY
  DECOMPOSITION::role_structure_mapping+component_identification+boundary_detection+interface_definition
  FREQUENCY::occurrence_counting+cross_role_validation+reuse_quantification+statistical_validation
  ORTHOGONALITY::dimension_independence+conflict_detection+compatibility_matrix+composition_validation
  MODULARITY::reusability_score+coupling_analysis+cohesion_measurement+complexity_assessment

PATTERN_CATEGORIZATION_FRAMEWORK:
  UNIVERSAL_PATTERNS::[all_role_requirements, foundational_directives, quality_standards]
  COGNITIVE_PATTERNS::[thinking_styles, decision_frameworks, processing_methods]
  OPERATIONAL_PATTERNS::[phase_contexts, seniority_modifiers, mode_variations]
  CAPABILITY_PATTERNS::[domain_expertise, tool_knowledge, methodology_skills]
  OUTPUT_PATTERNS::[response_structures, format_templates, communication_styles]

EXTRACTION_METHODOLOGY:
  EVIDENCE_REQUIREMENTS::[3+_independent_occurrences, clear_boundaries, generic_applicability]
  QUALITY_GATES::≥3_contexts+unambiguous_scope+explicit_conflicts+20-50_lines+≥0.7_modularity
  VALIDATION_PROCESS::[count_occurrences→validate_reusability→document_conflicts→extract_generic_form]

CONFLICTS::[theoretical_patterns, role_specific_content, single_use_components]
